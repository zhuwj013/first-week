#! /bin/bash
####################################################################
#$()：执行括号里的命令，并且将命令的输出捕获到变量中
####################################################################
input="hello,geocosl"
echo "$input"
echo "方法1："
echo "$input" | cut -d',' -f1
echo "方法2："
echo "${input%,*}" 
echo "方法3："
echo "$input" | awk -F',' '{print $1}'
echo "方法4："
echo "$input" | sed 's/,.*//'
