#! /bin/bash
####################################################################
#$()：执行括号里的命令，并且将命令的输出捕获到变量中
####################################################################
input="hello,geocosl"
echo "$input"
#################################################################### -d',':指定分隔符为逗号
echo "方法1："
echo "$input" | cut -d',' -f1
####################################################################%：从末尾开始删除最短匹配
echo "方法2："
echo "${input%,*}" 
####################################################################awk：文本处理工具； -F','：设置字段分隔符为逗号；打印第一个字段
echo "方法3："
echo "$input" | awk -F',' '{print $1}'
####################################################################,.*：匹配从第一个逗号开始的所有内容
echo "方法4："
echo "$input" | sed 's/,.*//'
